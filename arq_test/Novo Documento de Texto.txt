select p.colaborador, p.pergunta as Pergunta,
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta=1 and colaborador=p.colaborador) as '1 - Excelente',
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta=2 and colaborador=p.colaborador) as '2 - Muito Bom',
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta=3 and colaborador=p.colaborador) as '3 - Bom',
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta=4 and colaborador=p.colaborador) as '4 - Regular',
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta=5 and colaborador=p.colaborador) as '5 - Ruim',
(select count(resposta) from pesquisa where pergunta=p.pergunta and resposta between 1 and 5 and colaborador=p.colaborador) as 'TOTAL'
from pesquisa as p where data between '2016.01.01' and '2016.01.15'
group by colaborador, pergunta;
