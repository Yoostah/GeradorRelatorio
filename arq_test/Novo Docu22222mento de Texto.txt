select p.pergunta as Pergunta,
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and resposta=1 and data between $P{data_inicial} and $P{data_final}) as '1 - Excelente',
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and resposta=2 and data between $P{data_inicial} and $P{data_final}) as '2 - Muito Bom',
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and resposta=3 and data between $P{data_inicial} and $P{data_final}) as '3 - Bom',
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and resposta=4 and data between $P{data_inicial} and $P{data_final}) as '4 - Regular',
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and resposta=5 and data between $P{data_inicial} and $P{data_final}) as '5 - Ruim',
(select count(resposta) from pesquisa where pergunta=p.pergunta and colaborador= $P{colaborador} and data between $P{data_inicial} and $P{data_final} and (resposta=1 or resposta=2 or resposta=3 or resposta=4 or resposta=5)) as 'TOTAL'
from pesquisa as p where colaborador=$P{colaborador} and data between $P{data_inicial} and $P{data_final}
group by pergunta;




**********




union

select 'TOTAL', (select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and resposta=1 and data between $P{data_inicial} and $P{data_final}) as '1',
(select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and resposta=2 and data between $P{data_inicial} and $P{data_final}) as '2',
(select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and resposta=3 and data between $P{data_inicial} and $P{data_final}) as '3',
(select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and resposta=4 and data between $P{data_inicial} and $P{data_final}) as '4',
(select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and resposta=5 and data between $P{data_inicial} and $P{data_final}) as '5',
(select count(resposta) from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador} and (resposta=1 or resposta=2 or resposta=3 or resposta=4 or resposta=5) and data between $P{data_inicial} and $P{data_final}) as 'TOTAL' from pesquisa where pesquisa= $P{pesquisa} and colaborador= $P{colaborador}
group by 'TOTAL';
